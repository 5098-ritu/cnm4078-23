DECIMAL TO BINARY


#include < stdio.h >
#include < conio.h >
int main()
{
int n,rem,b=0,base=1;
printf("Enter a decimal number:");
scanf("%d",&n);
while(n>0)
{
rem=n%2;
b=b+rem*base;
n=n/2;
base=base*10;
}
printf("Binary number is:%d",b);
getch();
}

Output





Enter a decimal number: 9
Binary number is: 1001

BINARY TO DECIMAL



#include < stdio.h >
#include < conio.h >
int main()
{
int b,dec=0,i=1,rem;
printf("Enter a binary number:");
scanf("%d",&b);
while(b!=0)
{
rem=b%10;
dec=dec+rem*i;
i=i*2;
b=b/10;
}
printf("decimal number is:%d",dec);
getch();
}

Output


Enter a binary number: 1001
Decimal number is :9

#include < stdio.h >
#include < math.h >
void printfbinary(float number)
{
unsigned int binary=(unsigned int)&number;
//print the binary repersentation
for(int i=sizeof (float)*8-1;i>=0;i--)
{
printf("%d",(*binary >> i)&1);
if(i%8==0)
{
printf(" ");
}
}
printf("\n");
}

{

ADD/SUB/Multiplication/Division


 
#include < stdio.h >

void add(double a, double b);
void subtract(double a, double b);
void multiply(double a, double b);
void divide(double a, double b);

int main() {
    int choice;
    double num1, num2;

    do {
       
        printf("Arithmetic Operations Menu\n");
        printf("1. Addition\n");
        printf("2. Subtraction\n");
        printf("3. Multiplication\n");
        printf("4. Division\n");
        printf("5. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        if (choice >= 1 && choice <= 4) {
          
            printf("Enter two numbers: ");
            scanf("%lf %lf", &num1, &num2);
        }

       
        switch (choice) {
            case 1:
                add(num1, num2);
                break;
            case 2:
                subtract(num1, num2);
                break;
            case 3:
                multiply(num1, num2);
                break;
            case 4:
                divide(num1, num2);
                break;
            case 5:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice. Please try again.\n");
                break;
        }
        printf("\n");
    } while (choice != 5);

    return 0;
}
void add(double a, double b) {
    printf("Result: %.2lf\n", a + b);
}

void subtract(double a, double b) {
    printf("Result: %.2lf\n", a - b);
}

void multiply(double a, double b) {
    printf("Result: %.2lf\n", a * b);
}

void divide(double a, double b) {
    if (b != 0) {
        printf("Result: %.2lf\n", a / b);
    } else {
        printf("Error: Division by zero is not allowed.\n");
    }
}
