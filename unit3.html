LANGRANGE INTERPOLATION


 #include < stdio.h >

        double lagrangeInterpolation(double x[], double y[], int n, double xValue) {
            double result = 0.0;
        
            for (int i = 0; i < n; i++) {
                double term = y[i];
                for (int j = 0; j < n; j++) {
                    if (j != i) {
                        term *= (xValue - x[j]) / (x[i] - x[j]);
                    }
                }
                result += term;
            }
        
            return result;
        }
        
        int main() {
            int n;
            double x[10], y[10], xValue;
        
            printf("Enter the number of data points: ");
            scanf("%d", &n);
            
            printf("Enter the x and y values:\n");
            for (int i = 0; i < n; i++) {
                printf("x[%d]: ", i);
                scanf("%lf", &x[i]);
                printf("y[%d]: ", i);
                scanf("%lf", &y[i]);
            }
        
            printf("Enter the value of x to interpolate: ");
            scanf("%lf", &xValue);
        
            double interpolatedValue = lagrangeInterpolation(x, y, n, xValue);
            
            printf("The interpolated value at x = %.2f is %.6f\n", xValue, interpolatedValue);
        
            return 0;
        }

NEWTON FORWARD 


 
  #include < stdio.h >

        #define MAX 10
        
        void computeForwardDifference(float x[], float y[][MAX], int n) {
            for (int j = 1; j < n; j++) {
                for (int i = 0; i < n - j; i++) {
                    y[i][j] = y[i + 1][j - 1] - y[i][j - 1];
                }
            }
        }
        
        float newtonForwardInterpolation(float x[], float y[][MAX], int n, float value) {
            float h = x[1] - x[0];
            float result = y[0][0];
            float u = (value - x[0]) / h;
            
            for (int i = 1; i < n; i++) {
                float term = 1;
                for (int j = 0; j < i; j++) {
                    term *= (u - j) / (i);
                }
                result += term * y[0][i];
            }
        
            return result;
        }
        
        int main() {
            float x[MAX], y[MAX][MAX];
            int n;
            float value;
        
            printf("Enter the number of data points: ");
            scanf("%d", &n);
        
            printf("Enter the x values:\n");
            for (int i = 0; i < n; i++) {
                printf("x[%d]: ", i);
                scanf("%f", &x[i]);
            }
        
            printf("Enter the corresponding y values:\n");
            for (int i = 0; i < n; i++) {
                printf("y[%d]: ", i);
                scanf("%f", &y[i][0]);
            }
        
            computeForwardDifference(x, y, n);
        
            printf("Enter the value of x to interpolate: ");
            scanf("%f", &value);
        
            float interpolatedValue = newtonForwardInterpolation(x, y, n, value);
        
            printf("The interpolated value at x = %.2f is %.6f\n", value, interpolatedValue);
        
            return 0;

NEWTON BACKWARD 


 #include < stdio.h >

        #define MAX 10
        
        void computeBackwardDifference(float x[], float y[][MAX], int n) {
            for (int j = 1; j < n; j++) {
                for (int i = n - 1; i >= j; i--) {
                    y[i][j] = y[i][j - 1] - y[i - 1][j - 1];
                }
            }
        }
        
        float newtonBackwardInterpolation(float x[], float y[][MAX], int n, float value) {
            float h = x[1] - x[0];
            float result = y[n - 1][0];
            float u = (value - x[n - 1]) / h;
        
            for (int i = 1; i < n; i++) {
                float term = 1;
                for (int j = 0; j < i; j++) {
                    term *= (u + j) / (i);
                }
                result += term * y[n - 1 - i][i];
            }
        
            return result;
        }
        
        int main() {
            float x[MAX], y[MAX][MAX];
            int n;
            float value;
        
            printf("Enter the number of data points: ");
            scanf("%d", &n);
        
            printf("Enter the x values:\n");
            for (int i = 0; i < n; i++) {
                printf("x[%d]: ", i);
                scanf("%f", &x[i]);
            }
        
            printf("Enter the corresponding y values:\n");
            for (int i = 0; i < n; i++) {
                printf("y[%d]: ", i);

            }
        
            computeBackwardDifference(x, y, n);
        
            printf("Enter the value of x to interpolate: ");

NEWTON DIVIDED


  #include < stdio.h >

        #define MAX 10
        
        void computeDividedDifferences(float x[], float y[][MAX], int n) {
            for (int j = 1; j < n; j++) {
                for (int i = 0; i < n - j; i++) {
                    y[i][j] = (y[i + 1][j - 1] - y[i][j - 1]) / (x[i + j] - x[i]);
                }
            }
        }
        
        float newtonDividedDifferenceInterpolation(float x[], float y[][MAX], int n, float value) {
            float result = y[0][0];
            float product;
        
            for (int i = 1; i < n; i++) {
                product = 1;
                for (int j = 0; j < i; j++) {
                    product *= (value - x[j]);
                }
                result += product * y[0][i];
            }
        
            return result;
        }
        
        int main() {
            float x[MAX], y[MAX][MAX];
            int n;
            float value;
        
            printf("Enter the number of data points: ");
            scanf("%d", &n);
        
            printf("Enter the x values:\n");
            for (int i = 0; i < n; i++) {
                printf("x[%d]: ", i);
                scanf("%f", &x[i]);
            }
        
            printf("Enter the corresponding y values:\n");
            for (int i = 0; i < n; i++) {
                printf("y[%d]: ", i);

            }
        
            computeDividedDifferences(x, y, n);
        
            printf("Enter the value of x to interpolate: ");

TRAPEROIDAL RULE



#include < stdio.h >

        // Function to be integrated
        float f(float x) {
            return x * x; // Example function: f(x) = x^2
        }
        
        // Trapezoidal rule implementation
        float trapezoidal(float a, float b, int n) {
            float h = (b - a) / n; // Width of each trapezoid
            float sum = 0.5 * (f(a) + f(b)); // Sum of the first and last terms
        
            for (int i = 1; i < n; i++) {
                sum += f(a + i * h); // Add the middle terms
            }
        
            return sum * h; // Final result
        }
        
        int main() {
            float a, b; // Integration limits
            int n; // Number of trapezoids
        
            printf("Enter the lower limit (a): ");
            scanf("%f", &a);
            printf("Enter the upper limit (b): ");
            scanf("%f", &b);
            printf("Enter the number of trapezoids (n): ");
            scanf("%d", &n);
        
            float result = trapezoidal(a, b, n);
            printf("The approximate integral from %.2f to %.2f is %.6f\n", a, b, result);
        
            return 0;
        }

SIMPSON'S 1/3rd RULE



 #include < stdio.h >

        float f(float x) {
            return x * x; 
        }
        
        float simpsons13(float a, float b, int n) {
            if (n % 2 != 0) {
                printf("n must be even.\n");
                return 0;
            }
        
            float h = (b - a) / n;
            float sum = f(a) + f(b);
        
            for (int i = 1; i < n; i++) {
                if (i % 2 == 0) {
                    sum += 2 * f(a + i * h);
                } else {
                    sum += 4 * f(a + i * h);
                }
            }
        
            return sum * (h / 3);
        }
        
        int main() {
            float a, b;
            int n;
        
            printf("Enter the lower limit (a): ");
            scanf("%f", &a);
            printf("Enter the upper limit (b): ");
            scanf("%f", &b);
            printf("Enter the number of segments (n, must be even): ");
            scanf("%d", &n);
        
            float result = simpsons13(a, b, n);
            printf("The approximate integral from %.2f to %.2f is %.6f\n", a, b, result);
        
            return 0;
        }

        float simpsons38(float a, float b, int n) {
            if (n % 3 != 0) {
                printf("n must be a multiple of 3.\n");
                return 0;

            float h = (b - a) / n;
