SECANT METHOD


 
#include < stdio.h >
#include < math.h >
int main() {
 double x0, x1, tol, x2;
 int max_iter, iter = 0;
 // Function definition inline
 double f(double x) {
 return x * x * x - x - 2; // Example function: x^3 - x - 2
 }
 // Get user input
 printf("Enter the initial guess x0: ");
 scanf("%lf", &x0);
 
 printf("Enter the second guess x1: ");
 scanf("%lf", &x1);
 
 printf("Enter the tolerance: ");
 scanf("%lf", &tol);
 
 printf("Enter the maximum number of iterations: ");
 scanf("%d", &max_iter);
 printf("Iteration\t x0\t\t x1\t\t x2\t\t f(x2)\n");
 while (iter < max_iter) {
 double f_x0 = f(x0);
 double f_x1 = f(x1);
 // Check for division by zero
 if (f_x1 - f_x0 == 0) {
 printf("Error: Division by zero in the secant method.\n");
 return 1; // Exit with an error code
 }
 // Secant method formula
 x2 = x1 - f_x1 * (x1 - x0) / (f_x1 - f_x0);
 // Print the current iteration results
 printf("%d\t\t %.6f\t %.6f\t %.6f\t %.6f\n", iter + 1, x0, x1, x2, f(x2));
 // Check if the result is within the tolerance
 if (fabs(f(x2)) < tol || fabs(x2 - x1) < tol) {
 printf("Root found: %.6f\n", x2);
 return 0; // Successful exit
 }
 // Update the points for the next iteration
 x0 = x1;
 x1 = x2;
 iter++;
 }
 printf("Method did not converge after %d iterations.\n", max_iter);
 return 1; // Exit with an error code if not converged
}

NEWTON-RAPHSON METHOD




#include < stdio.h >
#include < math.h >
int main() {
 double x0, x1, tol;
 int max_iter, iter = 0;
 // Function to compute f(x) = x^3 - x - 2
 double f(double x) {
 return x * x * x - x - 2;
 }
 // Function to compute f'(x) = 3*x^2 - 1
 double f_prime(double x) {
 return 3 * x * x - 1;
 }
 // Get user input
 printf("Enter the initial guess x0: ");
 scanf("%lf", &x0);
 
 printf("Enter the tolerance: ");
 scanf("%lf", &tol);
 
 printf("Enter the maximum number of iterations: ");
 scanf("%d", &max_iter);
 printf("Iteration\t x0\t\t x1\t\t f(x1)\n");
 while (iter < max_iter) {
 double f_x0 = f(x0);
 double f_prime_x0 = f_prime(x0);
 // Check for division by zero
 if (f_prime_x0 == 0) {
 printf("Error: Division by zero in the Newton-Raphson method.\n");
 return 1; // Exit with an error code
 }
 // Newton-Raphson formula
 x1 = x0 - f_x0 / f_prime_x0;
 // Print the current iteration results
 printf("%d\t\t %.6f\t %.6f\t %.6f\n", iter + 1, x0, x1, f(x1));
 // Check if the result is within the tolerance
 if (fabs(f(x1)) < tol) {
 printf("Root found: %.6f\n", x1);
 return 0; // Successful exit
 }
 // Update the point for the next iteration
 x0 = x1;
 iter++;
 }
 printf("Method did not converge after %d iterations.\n", max_iter);
 return 1; // Exit with an error code if not converged
}

BIRGE-VIETA METHOD



#include < stdio.h >
#include < math.h >
int main() {
 double a, b, c, tol;
 int max_iter, iter = 0;
 double x0 = 0.0; // Initial guess
 double x1; // Next approximation
 double residual;
 // Get polynomial coefficients
 printf("Enter the coefficients for the polynomial ax^2 + bx + c:\n");
 printf("Enter coefficient a: ");
 scanf("%lf", &a);
 printf("Enter coefficient b: ");
 scanf("%lf", &b);
 printf("Enter coefficient c: ");
 scanf("%lf", &c);
 // Get tolerance and maximum number of iterations
 printf("Enter the tolerance: ");
 scanf("%lf", &tol);
 
 printf("Enter the maximum number of iterations: ");
 scanf("%d", &max_iter);
 // Print the header for the iterations
 printf("Iteration\t x0\t\t Residual\n");
 while (iter < max_iter) {
 // Evaluate the polynomial at x0
 residual = a * x0 * x0 + b * x0 + c;
 // Check for convergence
 if (fabs(residual) < tol) {
 printf("Root found: %.6f\n", x0);
 return 0; // Successful exit
 }
 // Derivative of the polynomial
 double f_prime_x0 = 2 * a * x0 + b;
 // Check for division by zero
 if (f_prime_x0 == 0) {
 printf("Error: Division by zero in the method.\n");
 return 1; // Exit with an error code
 }
 // Update the approximation
 x1 = x0 - residual / f_prime_x0;
 printf("%d\t\t %.6f\t %.6f\n", iter + 1, x0, residual);
 // Prepare for the next iteration
 x0 = x1;
 iter++;
 }
 printf("Method did not converge after %d iterations.\n", max_iter);
 return 1; // Exit with an error code if not converged
}

GAUSS-ELIMINATION METHOD



#include < stdio.h >
#define N 3 // Number of equations
int main() {
 int i, j, k;
 float A[N][N+1] = {
 {2, -1, 1, 3},
 {1, 2, 3, 8},
 {3, 1, -1, 3}
 };
 // Forward Elimination
 for (i = 0; i < N; i++) {
 // Pivot: Normalize the diagonal element
 float pivot = A[i][i];
 for (j = 0; j <= N; j++) {
 A[i][j] /= pivot;
 }
 // Eliminate entries below the pivot
 for (k = 0; k < N; k++) {
 if (k != i) {
 float factor = A[k][i];
 for (j = 0; j <= N; j++) {
 A[k][j] -= factor * A[i][j];
 }
 }
 }
 }
 // Print the solution
 printf("Solutions:\n");
 for (i = 0; i < N; i++) {
 printf("x%d = %.2f\n", i + 1, A[i][N]);
 }
 return 0;
}

GAUSS-JORDEN METHOD


#include < stdio.h >
int main() {
 int n;
 
 printf("Enter the number of variables: ");
 scanf("%d", &n);
 float matrix[n][n+1];
 // Input the augmented matrix
 printf("Enter the augmented matrix:\n");
 for (int i = 0; i < n; i++) {
 for (int j = 0; j <= n; j++) {
 scanf("%f", &matrix[i][j]);
 }
 }
 // Gauss-Jordan elimination
 for (int i = 0; i < n; i++) {
 // Make the diagonal contain all 1s
 for (int j = i+1; j < n; j++) {
 float ratio = matrix[j][i] / matrix[i][i];
 for (int k = 0; k <= n; k++) {
 matrix[j][k] -= ratio * matrix[i][k];
 }
 }
 }
 // Back-substitution to make the upper triangular matrix diagonal
 for (int i = n-1; i >= 0; i--) {
 for (int j = 0; j < i; j++) {
 float ratio = matrix[j][i] / matrix[i][i];
 for (int k = 0; k <= n; k++) {
 matrix[j][k] -= ratio * matrix[i][k];
 }
 }
 }
 // Print the solution
 printf("Solution:\n");
 for (int i = 0; i < n; i++) {
 printf("x%d = %.2f\n", i+1, matrix[i][n] / matrix[i][i]);
 }
 return 0;
}

GAUSS-SEIDEL METHOD



#include < stdio.h >

#define MAX 10
#define TOLERANCE 0.0001

void gaussSeidel(float a[MAX][MAX], float b[MAX], float x[MAX], int n) {
    float temp[MAX];
    int i, j;
    int iterations = 0;

    do {
        for (i = 0; i < n; i++) {
            temp[i] = x[i]; // Store the current value of x[i]
            x[i] = b[i];    // Start with the right-hand side value

            for (j = 0; j < n; j++) {
                if (j != i) {
                    x[i] -= a[i][j] * x[j]; // Subtract the known values
                }
            }
            x[i] /= a[i][i]; // Divide by the diagonal element
        }

        // Display the results after each iteration
        printf("Iteration %d: ", ++iterations);
        for (i = 0; i < n; i++) {
            printf("x[%d] = %.6f ", i, x[i]);
        }
        printf("\n");

    } while (1); // Keep iterating until convergence is detected

    // Check for convergence
    for (i = 0; i < n; i++) {
        if (i == 0) {
            if (temp[i] - x[i] < TOLERANCE && temp[i] - x[i] > -TOLERANCE) continue;
        } else {
            if (temp[i] - x[i] < TOLERANCE && temp[i] - x[i] > -TOLERANCE) continue;
        }
        break; // Exit the loop if not converged
    }
}

int main() {
    float a[MAX][MAX], b[MAX], x[MAX];
    int n, i, j;

    printf("Enter the number of equations: ");
    scanf("%d", &n);

    printf("Enter the coefficients of the equations (matrix A):\n");
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            scanf("%f", &a[i][j]);
        }
    }

    printf("Enter the constants (vector B):\n");
    for (i = 0; i < n; i++) {
        scanf("%f", &b[i]);
    }

    // Initialize solution vector x with zeros
    for (i = 0; i < n; i++) {
        x[i] = 0.0;
    }

    gaussSeidel(a, b, x, n);

    // Display the final solution
    printf("\nFinal solution:\n");
    for (i = 0; i < n; i++) {
        printf("x[%d] = %.6f\n", i, x[i]);
    }

    return 0;
}
